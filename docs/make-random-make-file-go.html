乱数を生成してファイルに書き込み

  • Redis のデータ生成
package main

import (
    "fmt"
    "io/ioutil"
    "math/rand"
    "strconv"
    "time"
)

const alphanumeric = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ1234567890"

func randString(n int) string {
    rand.Seed(time.Now().UnixNano())
    b := make([]byte, n)
    for i := range b {
        b[i] = alphanumeric[rand.Intn(len(alphanumeric))]

    }
    return string(b)
}

func main() {
    // 500 KB
    num := 524288

    for i := 0; i < 10; i++ {
        data := randString(num)
        s := strconv.Itoa(i)
        err := ioutil.WriteFile(s, []byte(data), 0664)
        if err != nil {
            fmt.Println(err)
        }
    }

}
