SQLite

  • connect

    $ sqlite3

    • current database

      $ sqlite3 db.database

  • create database(db_test) and table(table_test)

    $ sqlite3 db.db_test 'create table table_test(id integer PRIMARY KEY AUTOINCREMENT, status integer, score integer)'

  • show tables

    sqlite> select name from sqlite_master where type = 'table';
